// api/[feature]-api.ts
import { apiClient } from '@/lib/api';

export async function fetchItems(params?: QueryParams): Promise<ApiListResponse<Item>> {
  const response = await apiClient.get('/items/', { params });
  return response.data;
}

export async function fetchItem(publicId: string, isDeleted = false): Promise<Item> {
  const params = isDeleted ? { is_deleted: 'true' } : {};
  const response = await apiClient.get(`/items/${publicId}/`, { params });
  return response.data.data;
}

export async function createItem(data: CreatePayload): Promise<Item> {
  const response = await apiClient.post('/items/', data);
  return response.data.data;
}

export async function updateItem(publicId: string, data: UpdatePayload): Promise<Item> {
  const response = await apiClient.patch(`/items/${publicId}/`, data);
  return response.data.data;
}

export async function deleteItem(publicId: string): Promise<void> {
  await apiClient.delete(`/items/${publicId}/`);
}

export async function reactivateItem(publicId: string): Promise<Item> {
  const response = await apiClient.post(`/items/${publicId}/reactivate/`);
  return response.data.data;
}

// hooks/mutations.ts
export function useCreateItem() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: createItem,
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['items'] });
    },
  });
}

export function useUpdateItem() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: ({ id, data }: { id: string; data: UpdatePayload }) => 
      updateItem(id, data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['items'] });
    },
  });
}

export function useDeleteItem() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: deleteItem,
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['items'] });
    },
  });
}

export function useReactivateItem() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: reactivateItem,
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['items'] });
    },
  });
}
